class AppStore extends Component {
  state = {
    activeTabId: 'SOCIAL', // Set initial active tab to SOCIAL
    searchInput: '',
  }

  onChangeSearchInput = event => {
    this.setState({searchInput: event.target.value})
  }

  clickActiveTab = selected => {
    this.setState({activeTabId: selected})
  }

  getFilteredList = () => {
    const {activeTabId, searchInput} = this.state

    return appsList.filter(app => {
      const isMatchingCategory = app.category === activeTabId
      const isMatchingSearch =
        searchInput === '' ||
        app.appName.toLowerCase().includes(searchInput.toLowerCase())

      return isMatchingCategory && isMatchingSearch
    })
  }

  render() {
    const {activeTabId, searchInput} = this.state
    const filteredList = this.getFilteredList()

    return (
      <div className="body-container">
        <h1 className="heading">App Store</h1>
        <div className="search-box">
          <img
            src="https://assets.ccbp.in/frontend/react-js/app-store/app-store-search-img.png"
            alt="search icon"
            className="search-icon"
          />
          <input
            type="search"
            placeholder="Search"
            value={searchInput}
            onChange={this.onChangeSearchInput}
          />
        </div>
        <ul className="tabs-container">
          {tabsList.map(tabDetails => (
            <TabItem
              key={tabDetails.tabId}
              tabDetails={tabDetails}
              clickActiveTab={this.clickActiveTab}
              isActive={activeTabId === tabDetails.tabId}
            />
          ))}
        </ul>
        <ul className="app-list-container">
          {filteredList.map(appDetails => (
            <AppItem key={appDetails.appId} appDetails={appDetails} />
          ))}
        </ul>
      </div>
    )
  }
}

export default AppStore

import './index.css'

const TabItem = props => {
  const {tabDetails, clickActiveTab, isActive} = props
  const {tabId, displayText} = tabDetails
  const onClickTabItem = () => {
    clickActiveTab(tabId)
  }

  const activeTabBtnClassName = isActive ? 'active-tab-btn' : ''

  return (
    <li className="tab-item-container ">
      <button
        type="button"
        className={`tab-btn ${activeTabBtnClassName}`}
        onClick={onClickTabItem}
      >
        {displayText}
      </button>
    </li>
  )
}

export default TabItem
